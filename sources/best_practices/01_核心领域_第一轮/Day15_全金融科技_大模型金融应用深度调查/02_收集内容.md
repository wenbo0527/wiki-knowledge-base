### 3.2 关键实现技术

**语义分块策略：**

```python
class FinancialChunker:
    """金融文档语义分块器"""
    
    def chunk(self, document):
        # 按章节语义切分
        sections = self.parse_sections(document)
        
        chunks = []
        for section in sections:
            # 保留上下文信息
            chunk = {
                'content': section.text,
                'metadata': {
                    'title': section.title,
                    'date': section.date,
                    'source': section.source,
                    'company': section.company,
                    'section_path': section.path
                }
            }
            chunks.append(chunk)
        
        return chunks
    
    def merge_small_chunks(self, chunks, min_tokens=500):
        """合并过小的chunk"""
        merged = []
        buffer = None
        
        for chunk in chunks:
            if buffer is None:
                buffer = chunk
            elif buffer['token_count'] < min_tokens:
                buffer = self.merge(buffer, chunk)
            else:
                merged.append(buffer)
                buffer = chunk
        
        if buffer:
            merged.append(buffer)
        return merged
```

**混合检索实现：**

```python
class HybridRetriever:
    def retrieve(self, query, top_k=10):
        # 1. 向量检索
        vector_results = self.vector_store.search(
            query, top_k=top_k*2
        )
        
        # 2. BM25关键词检索
        keyword_results = self.keyword_index.search(
            query, top_k=top_k*2
        )
        
        # 3. RRF融合
        fused = self.rrf_fusion(
            vector_results, keyword_results, k=60
        )
        
        # 4. 重排序
        reranked = self.reranker.rerank(query, fused[:top_k])
        
        return reranked
    
    def rrf_fusion(self, results_list, k=60):
        """倒数排序融合 (RRF)"""
        scores = defaultdict(float)
        
        for results in results_list:
            for rank, item in enumerate(results):
                scores[item['id']] += 1 / (k + rank + 1)
        
        return sorted(scores.items(), key=lambda x: -x[1])
```

```python
class FinancialMultiAgent:
    """金融多Agent协作系统"""
    
    def __init__(self):
        self.planner = PlannerAgent()      # 任务规划
        self.researcher = ResearchAgent()   # 信息收集
        self.analyst = AnalystAgent()      # 分析推理
        self.reviewer = ReviewerAgent()    # 结果审核
        self.reporter = ReporterAgent()    # 报告生成
    
    async def handle_request(self, user_request):
        # Step 1: 任务规划
        plan = await self.planner.create_plan(user_request)
        
        # Step 2: 并行执行子任务
        tasks = []
        for subtask in plan.subtasks:
            if subtask.type == 'research':
                tasks.append(self.researcher.execute(subtask))
            elif subtask.type == 'analysis':
                tasks.append(self.analyst.execute(subtask))
        
        results = await asyncio.gather(*tasks)
        
        # Step 3: 结果审核
        reviewed = await self.reviewer.validate(results)
        
        # Step 4: 报告生成
        report = await self.reporter.generate(plan, reviewed)
        
        return report
```
